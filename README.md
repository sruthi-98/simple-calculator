### Simple Calculator

A simple calculator with basic arithmetic functions

### Installation

~~~
npm install @sruthisuresh/simple-calculator
~~~

### Usage

~~~
const { add, subtract, divide, multiply } = require('@sruthisuresh/simple-calculator');

add(6, 3)
// returns 9

subtract(6, 3)
// returns 3

multiply(6, 3)
// returns 18

divide(6, 3)
// returns 2
~~~
